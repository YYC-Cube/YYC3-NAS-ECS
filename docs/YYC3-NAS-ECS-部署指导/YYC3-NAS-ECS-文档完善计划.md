### 4.2 API示例代码

#### 4.2.1 JavaScript/TypeScript

```typescript
// 获取系统状态
async function getSystemStats() {
  const response = await fetch('/api/v2/system/stats', {
    headers: {
      'Authorization': `Bearer ${token}`
    }
  });
  
  const data = await response.json();
  
  if (data.success) {
    console.log('系统状态:', data.data);
  } else {
    console.error('错误:', data.error);
  }
}

// 设置告警阈值
async function setAlertThreshold(thresholds) {
  const response = await fetch('/api/v2/system/alert', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(thresholds)
  });
  
  const data = await response.json();
  
  if (data.success) {
    console.log('告警阈值已设置:', data.data);
  } else {
    console.error('错误:', data.error);
  }
}
```

#### 4.2.2 Python

```python
import requests

# 获取系统状态
def get_system_stats():
    headers = {
        'Authorization': f'Bearer {token}'
    }
    
    response = requests.get('/api/v2/system/stats', headers=headers)
    data = response.json()
    
    if data['success']:
        print('系统状态:', data['data'])
    else:
        print('错误:', data['error'])

# 设置告警阈值
def set_alert_threshold(thresholds):
    headers = {
        'Authorization': f'Bearer {token}',
        'Content-Type': 'application/json'
    }
    
    response = requests.post('/api/v2/system/alert', 
                           headers=headers, 
                           json=thresholds)
    data = response.json()
    
    if data['success']:
        print('告警阈值已设置:', data['data'])
    else:
        print('错误:', data['error'])
```
